# NextAuth.js for authentication
npm install next-auth@latest

# Authentication providers
npm install @auth/prisma-adapter
npm install bcrypt
npm install --save-dev @types/bcrypt

# Prisma ORM
npm install prisma @prisma/client

# Initialize Prisma (run after installation)
npx prisma init

# After creating your schema
npx prisma generate
npx prisma db push


# Tailwind CSS (if not already included in your Next.js setup)
npm install -D tailwindcss postcss autoprefixer
npx tailwindcss init -p

# shadcn/ui components (these are already included in your project)
# You can add more components as needed with:
npx shadcn@latest add [component-name]

# For charts and data visualization
npm install recharts


# Form validation
npm install zod
npm install react-hook-form @hookform/resolvers

# Date formatting
npm install date-fns

# HTTP client
npm install axios

# UUID generation
npm install uuid
npm install --save-dev @types/uuid

# CSRF protection
npm install csrf
npm install --save-dev @types/csrf


# For API rate limiting
npm install @upstash/redis
npm install @upstash/ratelimit

# For image uploads and optimization
npm install sharp
npm install --save-dev @types/sharp

# For SEO optimization
npm install next-sitemap


# ESLint and Prettier for code quality
npm install --save-dev eslint eslint-config-next prettier

# Testing libraries
npm install --save-dev jest @testing-library/react @testing-library/jest-dom
